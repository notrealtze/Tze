local function isMoveAllowed(moveName)
    local filter = getgenv().TZE.Filter
    if filter.UseBlacklist then
        for _, v in ipairs(filter.Blacklist) do
            if v == moveName then return false end
        end
    end
    if filter.UseWhitelist then
        for _, v in ipairs(filter.Whitelist) do
            if v == moveName then return true end
        end
        return false
    end
    return true
end

local function loadAnim(path)
    local obj = AnimFolder
    for _, part in ipairs(path) do
        obj = obj:FindFirstChild(part)
        if not obj then return end
    end
    return obj
end

local function registerSpecial(path, config)
    local anim = loadAnim(path)
    if anim then SpecialAnims[anim.AnimationId] = config end
end

local function getPing()
    return Stats.Network.ServerStatsItem["Data Ping"]:GetValue()
end

local function getBlockDelay()
    local base = getgenv().TZE.Combat.BlockDelay
    local ping = getPing() / 1000
    local jitter = getgenv().TZE.Humanize.ReactionDelay and math.random(0, 150) / 1000 or 0
    return math.max(0, base - ping) + jitter
end

local function getRange(moveName)
    if getgenv().TZE.Combat.DynamicRange and moveName and MoveRanges[moveName] then
        return MoveRanges[moveName]
    end
    return getgenv().TZE.Combat.Range
end

local function fireM1()
    if not getgenv().TZE.Combat.AutoPunish then return end
    if not isAlive() then return end
    if m1Cooldown then return end
    local char = LocalPlayer.Character
    if not char then return end
    local moveset = char:GetAttribute("Moveset")
    if not moveset then return end
    local service = RS.Knit.Knit.Services:FindFirstChild(moveset .. "Service")
    if not service then return end
    m1Cooldown = true
    service.RE.Activated:FireServer(false)
    task.delay(2, function() m1Cooldown = false end)
end

local function setBlock(state)
    if blocking == state then return end
    blocking = state
    if state then
        BlockActivate:FireServer()
    else
        BlockDeactivate:FireServer()
    end
end

local function getTargetName(target)
    local player = Players:GetPlayerFromCharacter(target)
    if player then return player.Name end
    return target.Name
end

local function isLocalPlayer(target)
    if target == LocalPlayer.Character then return true end
    if target.Name == LocalPlayer.Name then return true end
    if Players:GetPlayerFromCharacter(target) == LocalPlayer then return true end
    return false
end

local function isFacing(target)
    local char = LocalPlayer.Character
    if not char then return false end
    local root = char:FindFirstChild("HumanoidRootPart")
    local troot = target:FindFirstChild("HumanoidRootPart")
    if not root or not troot then return false end
    local toTarget = (troot.Position - root.Position).Unit
    local threshold = math.cos(math.rad(getgenv().TZE.Combat.FacingAngle))
    return root.CFrame.LookVector:Dot(toTarget) > threshold
end

local function isFacingAway(target)
    local char = LocalPlayer.Character
    if not char then return false end
    local root = char:FindFirstChild("HumanoidRootPart")
    local troot = target:FindFirstChild("HumanoidRootPart")
    if not root or not troot then return false end
    local toUs = (root.Position - troot.Position).Unit
    local threshold = math.cos(math.rad(getgenv().TZE.Combat.FacingAngle))
    return troot.CFrame.LookVector:Dot(toUs) < -threshold
end

local function inRange(target, moveName)
    local char = LocalPlayer.Character
    if not char then return false end
    local root = char:FindFirstChild("HumanoidRootPart")
    local troot = target:FindFirstChild("HumanoidRootPart")
    if not root or not troot then return false end
    return (root.Position - troot.Position).Magnitude <= getRange(moveName)
end

local function turnToFace(target)
    local char = LocalPlayer.Character
    if not char then return end
    local root = char:FindFirstChild("HumanoidRootPart")
    local troot = target:FindFirstChild("HumanoidRootPart")
    if not root or not troot then return end
    local dir = (troot.Position - root.Position).Unit
    workspace.CurrentCamera.CFrame = CFrame.new(workspace.CurrentCamera.CFrame.Position, workspace.CurrentCamera.CFrame.Position + dir)
end

local function isMe(target)
    return target == LocalPlayer.Character
end

local function blockFor(duration, name)
    if not isMoveAllowed(name) then return end
    setBlock(true)
    task.wait(duration)
    setBlock(false)
end

local function listenServiceMulti(serviceName, handlers)
    local ok, re = pcall(function()
        return getService(serviceName):WaitForChild("Effects")
    end)
    if not ok then return end
    re.OnClientEvent:Connect(function(effect, ...)
        if not isAlive() then return end
        local handler = handlers[effect]
        if handler then handler(...) end
    end)
end

task.spawn(function()
    while isAlive() do
        if getgenv().TZE.Humanize.RandomBlock and not blocking then
            task.wait(math.random(3, 8))
            if not getgenv().TZE.Humanize.RandomBlock then continue end
            setBlock(true)
            task.wait(math.random(3, 8) / 10)
            setBlock(false)
        else
            task.wait(1)
        end
    end
end)

local function watchTarget(target)
    if Characters[target] then return end
    if isLocalPlayer(target) then return end
    local hum = target:FindFirstChildOfClass("Humanoid")
    if not hum then return end

    Characters[target] = true
    local animator = hum:WaitForChild("Animator")

    animator.AnimationPlayed:Connect(function(track)
        if not isAlive() then return end
        if isLocalPlayer(target) then return end

        local fpsOffset = dt > 0.05 and 0.1 or 0
        local animId = track.Animation.AnimationId

        if getgenv().TZE.Combat.BlockDash and DashIDs[animId] then
            if not isMoveAllowed("Chase") then return end
            if inRange(target, "Chase") then
                if fpsOffset > 0 then task.wait(fpsOffset) end
                setBlock(true)
                track.Stopped:Wait()
                setBlock(false)
            end
            return
        end

        local special = SpecialAnims[animId]
        if special then
            if not inRange(target, special.name) then return end
            if not isMoveAllowed(special.name) then return end

            if special.blockIfFacingAway then
                if not isFacingAway(target) then return end
                setBlock(true)
                track.Stopped:Wait()
                setBlock(false)
                fireM1()
                return
            end

            if special.punishMidAnim then
                task.wait(0.3)
                fireM1()
                return
            end

            if special.turnCamera then
                turnToFace(target)
                setBlock(true)
                track.Stopped:Wait()
                setBlock(false)
                fireM1()
                return
            end

            if special.fullDuration then
                setBlock(true)
                track.Stopped:Wait()
                setBlock(false)
                return
            end

            if special.holdExtra then
                setBlock(true)
                track.Stopped:Wait()
                task.wait(special.holdExtra)
                setBlock(false)
                fireM1()
                return
            end

            return
        end

        if not getgenv().TZE.Combat.Autoblock then return end
        if not BlockableIDs[animId] then return end
        if not inRange(target, track.Animation.Name) then return end
        if getgenv().TZE.Combat.FacingCheck and not isFacing(target) then return end
        if math.random(1, 100) > getgenv().TZE.Combat.BlockChance then return end
        if getgenv().TZE.Humanize.MissBlock and math.random(1, 10) == 1 then return end
        if not isMoveAllowed(track.Animation.Name) then return end

        local delay = getBlockDelay() + fpsOffset
        if delay > 0 then task.wait(delay) end
        setBlock(true)
        track.Stopped:Wait()
        setBlock(false)

        if getgenv().TZE.Humanize.AntiBait and not isFacing(target) then return end
        fireM1()
    end)

    hum.Died:Connect(function()
        if not isAlive() then return end
        Characters[target] = nil
    end)
end

local function processContainer(container)
    for _, child in ipairs(container:GetChildren()) do
        if child:IsA("Model") or child:IsA("Folder") then
            if child:FindFirstChildOfClass("Humanoid") then watchTarget(child) end
            processContainer(child)
        end
    end
end

local charContainer = workspace:WaitForChild("Characters")
processContainer(charContainer)

charContainer.ChildAdded:Connect(function(child)
    task.wait(0.1)
    if not isAlive() then return end
    if child:IsA("Model") or child:IsA("Folder") then
        if child:FindFirstChildOfClass("Humanoid") then watchTarget(child) end
        processContainer(child)
    end
end)
